"""Zero-Free-Parameter constants -- replayable verification block.

Definitions precede every check, so each minpoly/value row is *recomputed*
from a closed form over its generator rather than reported. Two hardening
rules beyond the first draft:

  1. Conjugate disambiguation by value, not by hand-written interval.
     A minimal polynomial is Galois-invariant: a minpoly match certifies
     the conjugate orbit, not the element. Each row therefore carries the
     Section-2 tabulated decimal as its branch pin, so the check fails loudly
     both on a conjugate swap and on any definition that drifts from the
     published table. GAP and CRITICAL are the rows whose conjugates are
     both positive reals (genuinely confusable):
         GAP:      phi^-4 ~ 0.14590   vs  phi^4  ~ 6.85410   (x^2-7x+1)
         CRITICAL: phi^2/3 ~ 0.87268  vs  phi^-2/3 ~ 0.12732 (9x^2-9x+1)

  2. GAP := phi^-4, and K is wired to it. The table defines K = sqrt(1-gap);
     this is the real 0.92418 only when gap = phi^-4. With gap = phi^4,
     1-gap < 0 and K is non-real. Defining K = sqrt(1-GAP) makes the two
     branches structurally inseparable -- they cannot silently disagree.

Deps: sympy.  Run: python3 zfp_constants_repro.py

#!/usr/bin/env python3
"""r(r)=r as the lattice seed: idempotents, (semi)lattices, grid retractions.

Three readings of r(r)=r, each verified rather than asserted:

  (1) idempotent ELEMENT    e * e = e          -> over a field: e in {0, 1}
  (2) idempotent OPERATION  a * a = a          -> (semi)lattice; meet/join on Z^2
  (3) idempotent MAP        r(r(x)) = r(x)     -> retraction / closure operator;
                                                  rounding R^n -> Z^n, projection P^2 = P

Reading (2) is the genuine algebraic seed of lattice theory (Birkhoff 1940,
Davey-Priestley 2002): a semilattice IS a set with one commutative, associative,
idempotent operation. Reading (3) is the bridge to GEOMETRIC grids (Z^n): an
idempotent retraction maps ambient space onto the grid.

Note (see write-up): the equation that produces phi is x^2 = x + 1, which is
NOT x^2 = x. The literal idempotent seed r(r)=r yields {0,1} and lattices; it
does not yield phi. The two are different fixed points and are kept separate.

Deps: sympy.  Run: python3 rrr_idempotent_lattice.py
"""

import itertools
from sympy import Symbol, solve, sqrt

_FAILS = []
def _chk(cond):                      # record a failed structural check; printed value unchanged
    if not cond:
        _FAILS.append(1)
    return cond

# --------------------------------------------------- (1) element idempotents over a field
def element_idempotents():
    e = Symbol("e")
    idem = solve(e**2 - e, e)        # e*e = e          (the literal r(r)=r as a number)
    golden = solve(e**2 - e - 1, e)  # e*e = e + 1       (a DIFFERENT equation -> phi)
    print("(1) element reading e*e = e over a field")
    print(f"    e^2 = e      -> e in {sorted(idem)}            (only trivial idempotents)")
    print(f"    e^2 = e + 1  -> e in {golden}   <- this is phi's equation, NOT r(r)=r")
    _chk(sorted(idem) == [0, 1])     # the literal seed yields exactly the trivial idempotents

# --------------------------------------------------- (2) idempotent operations -> lattice on Z^2
def meet(a, b):                      # greatest lower bound under componentwise <=
    return (min(a[0], b[0]), min(a[1], b[1]))

def join(a, b):                      # least upper bound
    return (max(a[0], b[0]), max(a[1], b[1]))

def lattice_laws(grid):
    pairs   = list(itertools.product(grid, repeat=2))
    triples = list(itertools.product(grid, repeat=3))
    idem  = _chk(all(meet(a, a) == a and join(a, a) == a for a in grid))
    comm  = _chk(all(meet(a, b) == meet(b, a) and join(a, b) == join(b, a) for a, b in pairs))
    assoc = _chk(all(meet(meet(a, b), c) == meet(a, meet(b, c)) for a, b, c in triples))
    absor = _chk(all(meet(a, join(a, b)) == a and join(a, meet(a, b)) == a for a, b in pairs))
    distr = _chk(all(meet(a, join(b, c)) == join(meet(a, b), meet(a, c)) for a, b, c in triples))
    print("(2) Z^2 grid as an order lattice (meet = componentwise min, join = max)")
    print(f"    idempotent  r(r)=r : {idem}")
    print(f"    commutative        : {comm}")
    print(f"    associative        : {assoc}")
    print(f"    absorption         : {absor}   <- upgrades two semilattices into a lattice")
    print(f"    distributive       : {distr}   <- Z^n grids are distributive lattices")

# --------------------------------------------------- (3) idempotent maps -> grid mappings
def round_vec(x):                    # nearest-integer retraction R^n -> Z^n
    return tuple(round(c) for c in x)

def proj_x(v):                       # linear projection onto the x-axis; P^2 = P
    return (v[0], 0)

def closure_downset(S, n):           # order-closure on the chain {0..n}: c(S) = {0..max S}
    return frozenset(range(max(S) + 1)) if S else frozenset()

def map_laws():
    pts = [(1.4, -0.6), (2.5, 3.49), (-0.5, 0.5), (10.2, -3.8)]
    r1 = [round_vec(p) for p in pts]
    r2 = [round_vec(p) for p in r1]          # rounding already-integers is fixed
    round_ok = _chk(r1 == r2)
    print("(3) idempotent maps as grid mappings")
    print(f"    rounding R^2 -> Z^2 : r(r(x)) == r(x) ? {round_ok}   (retraction onto the grid)")
    pidem = _chk(all(proj_x(proj_x(v)) == proj_x(v) for v in [(3, 5), (-2, 7), (0, -4)]))
    print(f"    projection P^2 = P  : {pidem}   (idempotent endomorphism of the grid)")
    base = frozenset({1, 4})
    c1 = closure_downset(base, 6)

"""phi-dynamics on the idempotent grid: the Fibonacci matrix Q acting on Z^2.

Extends rrr_idempotent_lattice.py. The idempotent seed r(r)=r gave the min/max
lattice on Z^2. The golden seed r^2 = r + 1 gives Q = [[1,1],[1,0]], an automorphism
of the grid whose eigen-geometry is phi. This module wires the two together and
certifies every claim two independent ways with residual 0, matching the
constants-table discipline.

Branch pin: Q's eigenvalues phi ~ 1.618 and psi = -1/phi ~ -0.618 are Galois
conjugates sharing minpoly x^2 - x - 1, so 'showing phi' must pin the dominant
root -- the same situation as GAP and CRITICAL in the constants table.

Honesty point (carried from the previous step): Q^2 = Q + I, so Q is NOT idempotent
and is NOT a lattice endomorphism of (Z^2, min, max). It IS monotone, and it IS a
lattice isomorphism onto the pushforward order. phi-structure and idempotent
structure stay distinct objects; Q relates one to a golden-transformed copy of the
other.

Deps: sympy, and rrr_idempotent_lattice.py alongside.
Run:  python3 rrr_phi_grid.py
"""

import itertools
from sympy import (Matrix, Rational, Symbol, eye, expand, fibonacci, lucas,
                   minpoly, simplify, sqrt, zeros)
from rrr_idempotent_lattice import meet, join          # reuse the idempotent ops

x    = Symbol("x")
PHI  = (1 + sqrt(5)) / 2
PSI  = (1 - sqrt(5)) / 2                                # = -1/PHI, the conjugate eigenvalue
Q    = Matrix([[1, 1], [1, 0]])
QI   = Q.inv()                                          # integer inverse (det Q = -1)
N    = 12                                               # power / orbit horizon
DEEP = 30                                               # deep index for ratio convergence
EPS  = Rational(1, 10**12).evalf(50)

# integer action on grid points (tuples)
def q_apply(v):   return (v[0] + v[1], v[0])            # Q (x,y)^T = (x+y, x)
def qi_apply(v):  return (v[1], v[0] - v[1])            # Q^-1 (x,y)^T = (y, x-y)

def ok(cond, label, detail=""):
    print(f"{'PASS' if cond else 'FAIL'}  {label:<44}{detail}")
    assert cond, label

# --------------------------------------------------- Q1. Q is a grid automorphism (GL2(Z))
def check_automorphism():
    ok(Q.det() == -1, "det Q = -1", "(=> Q in GL2(Z))")
    ok(all(e.is_integer for e in QI), "Q^-1 has integer entries", f"  Q^-1 = {QI.tolist()}")
    ok(all(qi_apply(q_apply((a, b))) == (a, b)
           for a in range(-3, 4) for b in range(-3, 4)), "Q^-1 . Q = id on a Z^2 patch")

# --------------------------------------------------- Q2. powers = Fibonacci matrix; det/Cassini
def check_fib_matrix():
    for n in range(1, N + 1):
        F = Matrix([[fibonacci(n + 1), fibonacci(n)], [fibonacci(n), fibonacci(n - 1)]])
        assert Q**n == F, f"Q^{n} != Fibonacci matrix"
    ok(True, "Q^n = [[F_{n+1},F_n],[F_n,F_{n-1}]]", f"  n=1..{N}, residual 0")
    cassini = all((Q**n).det() == (-1)**n
                  == fibonacci(n + 1) * fibonacci(n - 1) - fibonacci(n)**2
                  for n in range(1, N + 1))
    ok(cassini, "det(Q^n) = (-1)^n = Cassini identity", "  (two independent routes)")

# --------------------------------------------------- Q3. eigen-decomposition shows phi (pinned)
def check_eigen():
    cp = Q.charpoly(x).as_expr()
    ok(expand(cp - (x**2 - x - 1)) == 0, "charpoly(Q) = x^2 - x - 1")
    ok(expand(minpoly(PHI, x) - (x**2 - x - 1)) == 0, "minpoly(phi) = x^2 - x - 1",
       "  (psi shares it)")
    lo, hi = Rational(161, 100).evalf(50), Rational(162, 100).evalf(50)
    ok(lo < PHI.evalf(50) < hi, "branch pin: dominant eigenvalue",
       f"  phi={PHI.evalf(8)}  (psi={PSI.evalf(8)} excluded)")
    ratio = Rational(fibonacci(DEEP + 1), fibonacci(DEEP))      # independent route to phi
    ok(abs(ratio - PHI).evalf(50) < EPS, "F_{n+1}/F_n -> phi", f"  |.- phi| < 1e-12 at n={DEEP}")
    V, D = Matrix([[PHI, PSI], [1, 1]]), Matrix([[PHI, 0], [0, PSI]])
    ok(simplify(Q * V - V * D) == zeros(2), "Q V = V D  (eigen-decomposition)", "  residual 0")
    ok(simplify(PHI * PSI + 1) == 0, "eigenvectors orthogonal", "  (phi,1).(psi,1)=phi*psi+1=0")

# --------------------------------------------------- Q4. trace ladder = Lucas (bridge to table)
def check_trace_lucas():
    for n in range(1, N + 1):
        assert (Q**n).trace() == lucas(n) == simplify(PHI**n + PSI**n), f"trace n={n}"
    ok(True, "trace(Q^n) = L_n = phi^n + psi^n", f"  n=1..{N}, three routes")
    ok((Q**4).trace() == 7 == lucas(4), "trace(Q^4) = 7 = L_4",
       "  (the constants-table normalizer)")

# --------------------------------------------------- Q5. orbit of a grid point
def check_orbit():
    cur, orbit = (1, 0), []
    for n in range(1, N + 1):
        cur = q_apply(cur)
        orbit.append(cur)
        assert cur == (fibonacci(n + 1), fibonacci(n)), f"orbit n={n}"
        assert isinstance(cur[0], int) and isinstance(cur[1], int)        # stays on Z^2
    ok(True, "orbit Q^n.(1,0) = (F_{n+1},F_n) on Z^2", f"  {orbit[:6]} ...")
    slope = Rational(fibonacci(DEEP), fibonacci(DEEP + 1))
    ok(abs(slope - 1 / PHI).evalf(50) < EPS, "orbit slope -> 1/phi = tau",
       "  (direction -> eigenvector (phi,1))")

# --------------------------------------------------- Q6. induced map on the min/max lattice
def meet_Q(a, b):  return q_apply(meet(qi_apply(a), qi_apply(b)))          # pushforward meet
def join_Q(a, b):  return q_apply(join(qi_apply(a), qi_apply(b)))

def check_induced_lattice():
    P = [(i, j) for i in range(-2, 3) for j in range(-2, 3)]              # 5x5 patch of Z^2
    pairs = list(itertools.product(P, repeat=2))
    le = lambda a, b: a[0] <= b[0] and a[1] <= b[1]
    # (a) positive: Q is monotone for the product order (nonneg entries preserve the orthant)
    ok(all(le(q_apply(a), q_apply(b)) for a, b in pairs if le(a, b)),
       "Q monotone on (Z^2, <=)", "  a<=b => Qa<=Qb")
    # (b) negative: Q does NOT preserve the original min/max meet (explicit witness)
    a, b = (1, 0), (0, 1)
    ok(q_apply(meet(a, b)) != meet(q_apply(a), q_apply(b)), "Q does NOT preserve min/max meet",
       f"  Q(a^b)={q_apply(meet(a,b))} != {meet(q_apply(a),q_apply(b))}")
    # (c) Q^-1 not monotone => Q is not an order-iso of the original order
    ok(not le(qi_apply((0, 0)), qi_apply((0, 1))), "Q^-1 NOT monotone",
       "  (so Q is not an order-iso of <=)")
    # (d) positive: Q IS a lattice iso onto the pushforward order (residual 0, all pairs)
    ok(all(q_apply(meet(a, b)) == meet_Q(q_apply(a), q_apply(b)) and
           q_apply(join(a, b)) == join_Q(q_apply(a), q_apply(b)) for a, b in pairs),
       "Q: lattice iso onto pushforward", "  Q(a^b)=meet_Q(Qa,Qb)")
    ok(all(meet_Q(q_apply(a), q_apply(a)) == q_apply(a) for a in P),
       "pushforward meet_Q idempotent", "  meet_Q(a,a)=a  (r(r)=r survives transport)")
    # (e) the dividing line: Q is the golden object, not an idempotent
    ok(simplify(Q * Q - (Q + eye(2))) == zeros(2) and Q * Q != Q, "Q^2 = Q + I  and  Q^2 != Q",
       "  (phi-structure != idempotent structure)")

if __name__ == "__main__":
